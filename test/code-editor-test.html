Console Test Commands

Copy these to browser console with Crab-Agent extension active:

// Detect editor
window.__crabCodeEditor?.detect()

// Get code
window.__crabCodeEditor?.getCode()

// Set code
window.__crabCodeEditor?.setCode('console.log("hello")')

// Insert at cursor
window.__crabCodeEditor?.insertAtCursor('// inserted')

// Get language
window.__crabCodeEditor?.getLanguage()

// Set language
window.__crabCodeEditor?.setLanguage('python')

// Format (Monaco only)
window.__crabCodeEditor?.format()

// Select lines 1-5
window.__crabCodeEditor?.selectLines(1, 5)
